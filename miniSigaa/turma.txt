
package miniSigaa0;

import java.util.ArrayList;


public class Turma {
    
    private int tamanho;
    private String codigo;
    private String disciplina;
    private String professor;
    private ArrayList<Aluno> aluno;
            
    public int getTamanho() {
        return tamanho;
    }

    public void setTamanho(int tamanho) {
        this.tamanho = tamanho;
    }
    
    public ArrayList<Aluno> getAluno(String alunos) {
        return aluno;
    }

    /**
     *
     * @param alunos
     */
    public void setAluno(ArrayList alunos) {
        this.aluno = aluno;
    }

    public String getProfessor() {
        return professor;
    }

    public void setProfessor(ArrayList<Professor> pp) {
        this.professor = professor;
    }

    public String getCodigo() {
        return codigo;
    }

    public void setCodigo(String codigo) {
        this.codigo = codigo;
    }

    public String getDisciplina() {
        return disciplina;
    }

    public void setDisciplina(ArrayList<Disciplina> dd) {
        this.disciplina = disciplina;
    }       
    

    @Override
    public String toString() {
        return "Turma: " + codigo;
    }
    
    public Turma(String codigo, ArrayList<Aluno> aluno) {
        this.codigo = codigo;
        this.aluno = new ArrayList<>();
        /* Verifica se o Aluno já está no array de Alunos */
        aluno.stream().forEach((a) -> {
            if (!this.aluno.contains(a)) {
                this.aluno.add(a);
                a.addTurma(this);
            } else {
                System.out.println("o " + a + ", já está matriculado em uma turma");
            }
        });
    }

    Turma() {
        //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
    }

}
